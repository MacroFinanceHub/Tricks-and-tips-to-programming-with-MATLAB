FOR-LOOPS

%this is in fact done by a for-loop, only implicitly
u = [5 4 8 8 2];
v = [5 5 7 8 8];
w = u -v

%the same with an explicitly loop
for ii = 1_lenght(u)
    w(ii) = u(ii) - v(ii);
end
